    3. 绘制条形图：

      javascript
      var w = 500;
  var h = 100;
  var barPadding = 1;

  var dataset = [5, 10, 13, 19, 21, 25, 22, 18, 15, 13,
   11, 12, 15, 20, 18, 17, 16, 18, 23, 25];

  //Create SVG element
  var svg = d3.select("body")
   .append("svg")
   .attr("width", w)
   .attr("height", h);

  svg.selectAll("rect")
   .data(dataset)
   .enter()
   .append("rect")
   .attr("x", function (d, i) {
    return i * (w / dataset.length);
                  //做一个位置排布的功能
   })
   .attr("y", function (d) {
    return h - d * 4;
    //高度减数值就可以画出条形图，因为一开始那个是按照svg坐标系排的
   })
   .attr("width", w / dataset.length - barPadding)//留下了一些位置方便空格，宽度和数据数的个数再减去间距
   .attr("height", function (d) {
    return d * 4;//让他变得更高一点点
   });

    .attr("fill", function(d) {
           return "rgb(0, 0, " + Math.round(d * 10) + ")";
                 //简易的颜色生成方式，可以生成偏蓝的颜色
       });





           //除了常见的数据值，还可以给条形图加上标签的方法，让数据更加直观

           svg.selectAll("text")
   .data(dataset)
   .enter()
   .append("text")
   .text(function(d) {
     return d;
   })//返回数据，赋值到这个东西上面
  .attr("x", function(d, i) {
     return i * (w / dataset.length) + (w / dataset.length - barPadding) / 2;//每个条左边的位置值加上条形宽度的一半
   })
   .attr("y", function(d) {
     return h - (d * 4) + 14;
   })
   .attr("font-family", "sans-serif")
             .attr("text-anchor","middle")
   .attr("font-size", "11px")
   .attr("fill", "white");//加上一些css样式

  • 比例尺：

    1. 是什么：比例尺是一种把输入域映射成输出范围的函数，代表一种数学关系
    2. ​输入值域：damain ||| 输出值域：range
    3. 创建线性比例尺：
    javascript
    var xScale = d3.scale.linear()

xScale.domain([0, 100])
.range([0, 500]);
    1. 缩放散点图：
    javascript
     //Width and height
var w = 500;
var h = 100;
var barPadding = 20

var dataset = [
   [5, 20], [480, 90], [250, 50], [100, 33], [330, 95],
   [410, 12], [475, 44], [25, 67], [85, 21], [220, 88]
    ];

//定义两个比例尺，对应x和y轴
var xScale = d3.scaleLinear()
     .domain([0, d3.max(dataset, function(d) { return d[0]; })])
     .range([barPadding, w-barPadding]);

var yScale = d3.scaleLinear()
     .domain([0, d3.max(dataset, function(d) { return d[1]; })])
     .range([h-barPadding, barPadding]);

//创建svg图元
var svg = d3.select("body")
  .append("svg")
  .attr("width", w)
  .attr("height", h);

//添加散点图（就是那几个圆）
svg.selectAll("circle")
  .data(dataset)
  .enter()
  .append("circle")
  .attr("cx", function(d) {
    return xScale(d[0]);
  })
  .attr("cy", function(d) {
    return yScale(d[1]);
  })
  //不能直接用散点值，要用比例尺进行转化
  .attr("r", function(d) {
    return Math.sqrt(h - d[1]);
  });

//Create 文本标签
svg.selectAll("text")
  .data(dataset)
  .enter()
  .append("text")
  .text(function(d) {
    return d[0] + "," + d[1];
  })
  .attr("x", function(d) {
    return xScale(d[0]);
  })
  .attr("y", function(d) {
    return yScale(d[1]);
  })
  .attr("font-family", "sans-serif")
  .attr("font-size", "11px")
  .attr("fill", "red");

    1. 以上代码，你可以随意修改svg的大小，而图表中的大小会随之成比例缩放

    2. 其他方法：

    3. nice():告诉比例尺，讲range的值自动调整到最接近的整数，比如，如果range的值是[0, 100]，那么nice()会自动变成[0, 100]，如果range的值是[0, 100.5]，那么nice()会自动变成[0, 101]

    4. rangeRound():同nice()一样，但是会四舍五入,有利于取代有用的像素值

    5. clamp():如果输入值域的值超出了比例尺的输出值域，那么clamp()会返回输出值域的边界值，比如，如果输入值域的值是[0 , 100]，那么clamp()会返回0和100，如果输入值域的值是[0, 100.5]，那么clamp()会返回0和100

    6. 只要把这些方法连缀在比例尺后面就好了

    7. 其他比例尺：d3内置了很多比例尺

      javascript
       //Width and height
   	var w = 500;
   	var h = 300;
   	var padding = 30;
   	
   	//Dynamic, random dataset
   	var dataset = [];										
   	var numDataPoints = 50;							
   	var maxRange = Math.random() * 1000;						
   	for (var i = 0; i < numDataPoints; i++) {			
   		var newNumber1 = Math.floor(Math.random() * maxRange);
   		var newNumber2 = Math.floor(Math.random() * maxRange);
   		dataset.push([newNumber1, newNumber2]);				
   	}

   	//Create scale functions
   	var xScale = d3.scaleLinear()
   						 .domain([0, d3.max(dataset, function(d) { return d[0]; })])
   						 .range([padding, w - padding * 2]);

   	var yScale = d3.scaleLinear()
   						 .domain([0, d3.max(dataset, function(d) { return d[1]; })])
   						 .range([h - padding, padding]);

   	//Define X axis
   	var xAxis = d3.axisBottom()
   					  .scale(xScale)
   					  .ticks(5);

   	//Define Y axis
   	var yAxis = d3.axisLeft()
   					  .scale(yScale)
   					  .ticks(5);

   	//Create SVG element
   	var svg = d3.select("body")
   				.append("svg")
   				.attr("width", w)
   				.attr("height", h);

   	//Create circles
   	svg.selectAll("circle")
   	   .data(dataset)
   	   .enter()
   	   .append("circle")
   	   .attr("cx", function(d) {
   	   		return xScale(d[0]);
   	   })
   	   .attr("cy", function(d) {
   	   		return yScale(d[1]);
   	   })
   	   .attr("r", 2);
   	
   	//Create X axis
   	svg.append("g")
   		.attr("class", "x axis")
   		.attr("transform", "translate(0," + (h - padding) + ")")
   		.call(xAxis);
   	
   	//Create Y axis
   	svg.append("g")
   		.attr("class", "y axis")
   		.attr("transform", "translate(" + padding + ",0)")
   		.call(yAxis);



   	//On click, update with new data			
   	d3.select("p")
   		.on("click", function() {
     //点击更新图表
   	
   			var numValues = dataset.length;						 	
   			var maxRange = Math.random() * 1000;					
   			dataset = [];  						 				 		
   			for (var i = 0; i < numValues; i++) {				 		
   				var newNumber1 = Math.floor(Math.random() * maxRange);	
   				var newNumber2 = Math.floor(Math.random() * maxRange);	
   				dataset.push([newNumber1, newNumber2]);				
       //拿到新的随机数据
   			}
   			
   			//更新比例尺
   			xScale.domain([0, d3.max(dataset, function(d) { return d[0]; })]);
   			yScale.domain([0, d3.max(dataset, function(d) { return d[1]; })]);

   			//更新图上面的点
   			svg.selectAll("circle")
   			   .data(dataset)
   			   .transition()
   			   .duration(1000)		
   			   .attr("cx", function(d) {
   			   		return xScale(d[0]);
   			   })
   			   .attr("cy", function(d) {
   			   		return yScale(d[1]);
   			   });

   			//更新数轴
   			svg.select(".x.axis")

    1. json和geojson的区别和共同点：

      • json：json是javascript对象表示法，它描述了javascript对象，javascript对象是由键值对组成的，键是字符串，值可以是任意的javascript对象，包括数组、对象、字符串、数字、布尔值、null等。

      • geojson：geojson是geojson对象表示法，它描述了地理信息对象，地理信息对象是由键值对组成的，键是字符串，值可以是任意的geojson对象，包括数组、对象、字符串、数字、布尔值、null等。

      • json和geojson共同点：

        • 都是javascript对象表示法，都是由键值对组成的，键是字符串，值可以是任意的javascript对象，包括数组、对象、字符串、数字、布尔值、null等。

        经纬度 经度：东经为正，西经为负，单位为度，范围从0到180度。 纬度：北纬为正，南纬为负，单位为度，范围从0到90度。
